from flask import Flask, request, jsonify, session, render_template import sqlite3 app = Flask(__name__) app.secret_key = "super_secret_key" # ====================== # DB INIT # ====================== def init_db(): conn = sqlite3.connect("db.db") c = conn.cursor() c.execute(""" CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTOINCREMENT, username TEXT, password TEXT ) """) conn.commit() conn.close() init_db() # ====================== # REGISTER (test) # ====================== @app.route("/register", methods=["POST"]) def register(): data = request.get_json() u = data.get("username") p = data.get("password") conn = sqlite3.connect("db.db") c = conn.cursor() c.execute("INSERT INTO users (username, password) VALUES (?,?)", (u, p)) conn.commit() conn.close() return jsonify({"msg": "created"}) # ====================== # LOGIN # ====================== @app.route("/login", methods=["POST"]) def login(): data = request.get_json() u = data.get("username") p = data.get("password") conn = sqlite3.connect("db.db") c = conn.cursor() c.execute("SELECT * FROM users WHERE username=? AND password=?", (u, p)) user = c.fetchone() conn.close() if user: session["user"] = u return jsonify({"msg": "ok"}) return jsonify({"msg": "error"}) # ====================== # LOGIN PAGE # ====================== @app.route("/login") def login_page(): return render_template("login.html") # ====================== # HOME TEST # ====================== @app.route("/") def home(): if "user" not in session: return render_template("login.html") return "Logged in ✔" if __name__ == "__main__": app.run(debug=True)
